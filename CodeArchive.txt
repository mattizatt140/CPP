#include "stdafx.h"
#include <iostream>
using namespace std;

double avg(int _array[])
{
	int totalSum{};

	int sizeof_ = sizeof(_array);


	for (int i = 0; i < sizeof(_array); i++) {

		totalSum += _array[i];
	}
	
	double totalAvg = totalSum / sizeof(_array);
	
	return totalAvg;
}

int main()
{
	int newArray[] = { 1,2,3,4,5 };
	double newAvg = avg(newArray);
	cout << newAvg << endl;
}
